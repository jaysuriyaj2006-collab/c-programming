#include <stdio.h>
#include <stdlib.h>

struct Node {
    int doctorID;
    struct Node *prev;
    struct Node *next;
};

int main() {
    struct Node *head = NULL, *newNode = NULL, *last = NULL;
    int n, i, value;

    printf("Enter number of insertions: ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        newNode = (struct Node *)malloc(sizeof(struct Node));
        printf("Enter doctor ID: ");
        scanf("%d", &value);

        newNode->doctorID = value;

        if (head == NULL) {
            head = newNode;
            newNode->next = head;
            newNode->prev = head;
        } else {
            last = head->prev;

            newNode->next = head;
            newNode->prev = last;

            last->next = newNode;
            head->prev = newNode;

            head = newNode;   // update head
        }
    }

    // Display list
    printf("\nDoctor Rotation List:\n");
    last = head->prev;
    newNode = head;
    do {
        printf("%d <-> ", newNode->doctorID);
        newNode = newNode->next;
    } while (newNode != head);

    printf("(back to %d)", head->doctorID);

    return 0;
}

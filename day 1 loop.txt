#include<stdio.h>
int main(){
    int pin=2345;
    int userpin,attempt=3,rb,wd,amount=10000;
    while(attempt>0){
        printf("\nenter your pin:");
        scanf("%d",&userpin);
        if(userpin==pin){
            printf("\npin is correct");
            printf("\nenter amount to withdraw:");
            scanf("%d",&wd);
            if(wd<=amount){
                printf("\ntransaction successfull\n");
                rb=amount-wd;
                printf("\nremaining balance=%d",rb);
            }
            else{
                printf("\ntransaction failed");
            }
        }
        else{
            printf("\nincorrect pin");
            attempt--;
        }
    }
    return 0;
}

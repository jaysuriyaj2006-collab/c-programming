#include <stdio.h>
#include <stdlib.h>

struct Node {
    int songID;
    struct Node *next;
};

int main() {
    struct Node *head = NULL, *temp = NULL, *newNode = NULL;
    int n, i, value;

    printf("Enter number of songs to insert: ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        newNode = (struct Node *)malloc(sizeof(struct Node));

        printf("Enter song ID: ");
        scanf("%d", &value);

        newNode->songID = value;

        if (head == NULL) {
            head = newNode;
            newNode->next = head;   // circular link
        } else {
            temp = head;
            // move to last node
            while (temp->next != head) {
                temp = temp->next;
            }
            temp->next = newNode;
            newNode->next = head;   // maintain circularity
        }
    }

    // Display circular linked list
    printf("\nPlaylist:\n");
    temp = head;
    do {
        printf("%d -> ", temp->songID);
        temp = temp->next;
    } while (temp != head);

    printf("(back to %d)", head->songID);

    return 0;
}
